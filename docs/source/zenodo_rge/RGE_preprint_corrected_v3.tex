\section{Core Definitions}
\begin{definition}[Recursive subdivision]
Let $\Omega\subset\mathbb{R}^n$ be bounded. At depth $d$, a subdivision induces disjoint parts
$\{P_{j,d}\}_{j=1}^{N_d}$ with $\cup_j P_{j,d}=\Omega$.
\end{definition}

\begin{definition}[Depth entropy]
With probabilities
\[
p_j(d)=\frac{\mu(P_{j,d})}{\mu(\Omega)},
\]
define
\[
S(d):=-\sum_{j=1}^{N_d} p_j(d)\log_2 p_j(d).
\]
\end{definition}

\begin{definition}[Asymptotic slope-offset form]
When limits exist,
\[
S(d)=C\,d+h_\Omega+o(1),
\]
where $C$ is the growth slope and $h_\Omega$ is the shape-dependent offset.
\end{definition}

\section{Asymptotic Law for Depth Entropy}
\begin{theorem}[Main asymptotic law]
Assume:
\begin{enumerate}
    \item $\lim_{d\to\infty} (\log_2 N_d)/d = C$ exists,
    \item for $m=1,\dots,N_d$, probabilities admit
    \[
    p_{m,d}=\frac{1}{N_d}f\!\left(\frac{m+1/2}{N_d}\right)+o\!\left(\frac{1}{N_d}\right),
    \]
    with bounded measurable $f\ge0$, $\int_0^1 f(x)dx=1$.
\end{enumerate}
Then
\[
S(d)=\log_2 N_d-\int_0^1 f(x)\log_2 f(x)\,dx + o(1)
= C\,d + h_2(f) + o(1).
\]
\end{theorem}

\begin{proof}[Proof sketch]
Rewrite entropy as
\[
S(d)=\log_2 N_d - \sum_{m=1}^{N_d} p_{m,d}\log_2\big(p_{m,d}N_d\big).
\]
Under the density form, the second term is a Riemann sum converging to
$\int_0^1 f\log_2 f$.
\end{proof}

\section{Radial Refinements and Closed-Form Offsets}
\begin{theorem}[Universal radial-shell slope]
For radial shells with $N_d=2^d$, the slope is
\[
C=1.
\]
\end{theorem}

\begin{proof}
Immediate from $\log_2 N_d=\log_2 2^d=d$.
\end{proof}

\begin{proposition}[Disk offset]
For $f(r)=2r$ on $[0,1]$:
\[
h_{disk}=-\int_0^1 2r\log_2(2r)dr=-1+\frac{1}{2\ln2}\approx -0.2786524796.
\]
\end{proposition}

\begin{proposition}[Sphere offset]
For $f(r)=3r^2$ on $[0,1]$:
\[
h_{sphere}=-\int_0^1 3r^2\log_2(3r^2)dr=-\log_2 3+\frac{2}{3\ln2}\approx -0.6231658068.
\]
\end{proposition}

\begin{proposition}[Cone offset]
For $f(r)=6r(1-r)$ on $[0,1]$:
\[
h_{cone}=-\int_0^1 6r(1-r)\log_2\!\big(6r(1-r)\big)dr
=-\log_2 6+\frac{5}{3\ln2}\approx -0.1804707659.
\]
\end{proposition}

\section{Congruent Subdivision Slopes}
\begin{theorem}
If each part splits into $b$ congruent parts per depth, so $N_d=b^d$, then
\[
C=\log_2 b.
\]
\end{theorem}

\begin{proof}
$\log_2 N_d = \log_2 b^d = d\log_2 b$.
\end{proof}

\section{Finite-Depth Growth-Factor Phenomenon}
For the radial asymptotic form $S(d)=d+h_\Omega+o(1)$,
\[
\tilde r(d):=\frac{S(d+1)}{S(d)}=\frac{d+1+h_\Omega}{d+h_\Omega}+o(1/d)
=1+\frac{1}{d+h_\Omega}+o(1/d).
\]
Solving $\tilde r(d)=1.25$ gives
\[
d_{1.25}=4-h_\Omega.
\]
This is a finite-depth crossing law, not an asymptotic invariant.

\section{Independent-Refinement Additivity (Restricted Theorem)}
\begin{theorem}[Model-specific additivity]
Assume independent orthogonal product refinement across $k$ axes, each with branching
$b_i$ and no coupling in partition probabilities. Then
\[
C=\sum_{i=1}^{k}\log_2 b_i.
\]
\end{theorem}

\begin{proof}
Under product independence, total branching per depth is multiplicative:
$N_d=(\prod_i b_i)^d$. Therefore
\[
\frac{\log_2 N_d}{d}=\log_2\Big(\prod_i b_i\Big)=\sum_i \log_2 b_i.
\]
\end{proof}

\begin{remark}[General case]
Without explicit independence/product structure, additivity remains a conjecture.
\end{remark}
